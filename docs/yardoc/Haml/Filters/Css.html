def render_with_options(text, options)
  if options[:format] == :html5
    type = ''
  else
    type = " type=#{options[:attr_wrapper]}text/css#{options[:attr_wrapper]}"
  end

  "<style\#{type}>\n  /*<![CDATA[*/\n    \#{text.rstrip.gsub(\"\\n\", \"\\n    \")}\n  /*]]>*/\n</style>\n"
end
